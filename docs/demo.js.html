/**
 * Demo contract: increments a counter on state.
 * @param {{counter?: number}} state  Current contract state.
 * @param {{inc?: number}}  input  Input object with `inc` value.
 * @returns {{counter: number}} Updated state.
 */
module.exports = function (state, input) {
  if (!state.counter) state.counter = 0;
  if (input && input.inc) state.counter += input.inc;
  return state;
};
